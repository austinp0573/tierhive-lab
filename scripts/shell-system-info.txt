(
  echo "=== OPERATING SYSTEM & VERSION ==="
  cat /etc/alpine-release 2>/dev/null || cat /etc/os-release
  
  echo -e "\n=== KERNEL ARCHITECTURE ==="
  uname -a
  
  echo -e "\n=== HYPERVISOR / VIRTUAL HARDWARE DETECTED ==="
  cat /sys/devices/virtual/dmi/id/sys_vendor 2>/dev/null && echo -n "Vendor: " && cat /sys/devices/virtual/dmi/id/sys_vendor
  cat /sys/devices/virtual/dmi/id/product_name 2>/dev/null && echo -n "Product: " && cat /sys/devices/virtual/dmi/id/product_name
  cat /proc/cpuinfo | grep -E 'model name|vendor_id' | uniq
  
  echo -e "\n=== ACTIVE INITIALIZATION RUNLEVELS ==="
  rc-status -s 2>/dev/null || rc-status --all
  
  echo -e "\n=== LOADED KERNEL DRIVERS ==="
  lsmod
)
